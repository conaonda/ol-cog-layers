Side-by-Side Comparison

Vanilla OpenLayers

import GeoTIFF from 'ol/source/GeoTIFF';
import WebGLTileLayer from 'ol/layer/WebGLTile';

// UTM COG on EPSG:3857 view — misaligned or broken
const source = new GeoTIFF({
  sources: [{ url: skySatUrl, bands: [1, 2, 3] }],
});
const layer = new WebGLTileLayer({ source });

const map = new Map({
  target: 'map-vanilla',
  layers: [new TileLayer({ source: new OSM() }), layer],
  view: new View({ projection: 'EPSG:3857' }),
});

ol-cog-layers (affine)

import { createCOGLayer } from 'ol-cog-layers';

// Same COG — pixel-perfect on EPSG:3857
const { layer, extent } = await createCOGLayer({
  url: skySatUrl,
  viewProjection: 'EPSG:3857',
  projectionMode: 'affine',
});

const map = new Map({
  target: 'map-easy',
  layers: [new TileLayer({ source: new OSM() }), layer],
  view: new View({ projection: 'EPSG:3857' }),
});
map.getView().fit(extent);
