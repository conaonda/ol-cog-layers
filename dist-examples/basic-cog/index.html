import { Map, View } from 'ol';
import TileLayer from 'ol/layer/Tile';
import OSM from 'ol/source/OSM';
import { createCOGLayer } from 'ol-cog-layers';

// SkySat RGB (UTM) → EPSG:3857 via affine transform
const { layer, extent } = await createCOGLayer({
  url: 'https://storage.googleapis.com/.../SkySat_RGB.tif',
  viewProjection: 'EPSG:3857',
  projectionMode: 'affine',   // bypass OL reprojection
});

const map = new Map({
  target: 'map',
  layers: [new TileLayer({ source: new OSM() }), layer],
  view: new View({ projection: 'EPSG:3857' }),
});
map.getView().fit(extent);
